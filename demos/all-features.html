// Create store
const store = _.createStore({
    state: { count: 0 },
    mutations: {
        increment(state) { state.count++; },
        decrement(state) { state.count--; },
        reset(state) { state.count = 0; }
    }
});

// Use store

_('#decrement').on('click', () => store.commit('decrement'));
_('#undo').on('click', () => store.undo());
_('#redo').on('click', () => store.redo());

‚ö° Signals & Reactivity

SolidJS-inspired reactive primitives

Reactive Counter:

0
Double: 0
// Create signal
const [count, setCount] = _.signal(0);

// Computed value
const doubled = _.computed(() => count() * 2);

// Effect - auto runs when signal changes
_.effect(() => {
    _('#signal-count').text(count());
    _('#signal-double').text('Double: ' + doubled());
});

// Update signal
_('#signal-increment').on('click', () => {
    setCount(count() + 1);
});

// GET request
_('#fetch-users').on('click', () => {
    _.get('https://jsonplaceholder.typicode.com/users')
        .then(data => {
            _('#http-result').html(
                '<h5 class="font-bold mb-2">Users:</h5>' +
                data.slice(0, 3).map(u => 
                    `<div>${u.name} (${u.email})</div>`
                ).join('')
            );
        })
        .catch(err => {
            _('#http-result').html('Error: ' + err.message);
        });
});
