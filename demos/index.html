Quick Start

CDN (Recommended)

<script src="https://cdn.jsdelivr.net/gh/Yaka-UI-Labs/YakaJS@latest/dist/min.yaka.js"></script>

NPM

npm install yakajs

Simple Example

// DOM Ready
_(() => {
    // Select elements and manipulate
    _('#myButton').on('click', () => {
        _('#content').fadeIn(500);
    });
    
    // AJAX made simple
    _.get('/api/data').then(data => {
        _('#output').html(data);
    });
});
