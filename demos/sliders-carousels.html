üìö API Reference

Slider Options

{
    min: 0,              // Minimum value
    max: 100,            // Maximum value
    step: 1,             // Step increment
    value: 50,           // Initial value
    showTooltip: true,   // Show tooltip above slider
    showValue: true,     // Show value below slider
    onChange: (val) => {}// Callback on value change
}

Carousel Options

{
    auto: false,         // Auto-advance slides
    interval: 3000,      // Auto-advance interval (ms)
    showDots: true,      // Show dot indicators
    showArrows: true,    // Show arrow buttons
    startIndex: 0,       // Starting slide index
    transition: 'fade',  // 'fade', 'slide', or 'none'
    touch: true,         // Enable touch/swipe
    onChange: (idx) => {}// Callback on slide change
}
